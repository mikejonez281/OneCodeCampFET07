Check for git version = git --version

Configure Git = git config --global user.name and git config --global user.email

Initialize and empty repository = git init 

Check for the status of the files = git status

Add files to the staging area = git add .

Create a commit = git commit -m "This is the description"

Bypass/Skip Staging Area = git commit -a -m "Description"

U - Untracked
M - Modified
A - Added

Show History Author, Date, Description = git log
Show Description Only = git log --oneline

List Branches = git branch -a 

Add Branch = git branch [Description]

Switch Branches = git switch [Description]
